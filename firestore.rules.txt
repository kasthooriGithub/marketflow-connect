rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isSignedIn() {
      return request.auth != null;
    }
    
    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    match /orders/{orderId} {
      // Create: Allow if signed in and client_id matches user's UID
      allow create: if isSignedIn() && isOwner(request.resource.data.client_id);
      
      // Read: Allow if user is either the client OR the vendor for this order
      allow read: if isSignedIn() && (
        isOwner(resource.data.client_id) || 
        isOwner(resource.data.vendor_id)
      );
      
      // Update: 
      // Vendor: Allow changing status if they own the vendor_id
      allow update: if isSignedIn() && 
                    isOwner(resource.data.vendor_id) && 
                    request.resource.data.diff(resource.data).affectedKeys().hasOnly(['status', 'updated_at']);
    }
  }
}
